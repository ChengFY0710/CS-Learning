# 12 - 5 链表

## 01 创建链表
以创建一个含10个单位的链表。
```c
#include <stdio.h>
#include <stdlib.h>
#define LEN sizeof(struct number)
struct number{
    int num;
    struct number * next;
};
int main()
{
    struct number * head, * p1, * p2;
    head = p1 = p2 = (struct number *)malloc(LEN);
    for(int i=1; i<10; i++){
        p1 = (struct number *)malloc(LEN);
        p2->next = p1;
        p2 = p1; 
    }
    p2->next = NULL;

    return 0;
}
```

## 02 遍历链表

p1 = p2 = head;

## 03 释放链表
```c
p1 = p2 = head;
while(p1->next != NULL){
    p2 = p1->next;
    free(p1);
    p1 = p2;
}
```

## 04 链表的删除
假设要删除第4个节点
![12 - 5 - 1](/引用文件/12%20-%205%20-%201.png)
```c
p1 = p2 = head;
for(int i=1; i<3; i++){
    p1 = p1->next;
}
// 此时p1指向第3个节点
p2 = p1->next;  // 使得p2指向第4个节点
p1->next = p2->next; 
```
具体实例
```cpp
node* deleteElements(node*head, int v)
{
    if (head == NULL) return NULL;
    node *p1, *p2;
    p2 = head;
    while(p2->next != NULL){
        p1 = p2->next;
        if(p1->val == v){
            p2->next = p1->next;
        }
        else {
        p2 = p2->next;
        }
    }
    if(head->val == v){
        head = head->next;
    }
    return head;
}
```
>[!tip]
>链表删除时，如果需要删除，赋值后，**前指针不动，后指针后移**。

## 06 链表的交换
```cpp
node* createList()
{
    node *head, *p1, *p2;
    head = p1 = p2 = NULL;
    int num;
    scanf("%d", &num);
    if(num < 0) return head;
    else{
        head = p1 = p2 = (node *)malloc(sizeof(node));
        while(1){
            p2->val = num;
            scanf("%d", &num);
            if(num < 0){
                p2->next = NULL;
                return head;
            }
            p1 = (node *)malloc(sizeof(node));
            p2->next = p1;
            p2 = p1;
        }
    }
}

node* reverseList(node *head)
{
    node *p1, *p2, *p3;
    p1 = head;
    if(head == NULL) return NULL;
    p2 = p1->next;
    while(p2 != NULL){
        p3 = p2->next;
        p2->next = p1;
        p1 = p2;
        p2 = p3;
    }
    head->next = NULL;
    head = p1;
    return head;
}
```
>[!tip]
>链表的交换时，**使用p1指针指向当前项，p2指针指向下一项，由于需要倒置，故p2->next会破坏，故需要一开始用p3指针保留p2的下一项**。

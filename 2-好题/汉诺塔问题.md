```c
void move(char one, char two)
{
    printf("%c to %c", one, two);
}
void operate(int n, char one, char two, char three)
{
    if(n == 1) move(one, three);
    operate(n-1, one, three, two);
    move(one, three);
    operate(n-1, two, one, three);
}
```
